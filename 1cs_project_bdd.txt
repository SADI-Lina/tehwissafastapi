drop database tehwissa;
CREATE database tehwissa;
USE tehwissa;


CREATE TABLE region (
    id INT PRIMARY KEY AUTO_INCREMENT,
    designation TEXT
);

CREATE TABLE wilaya (
    code INT PRIMARY KEY,
    designation TEXT,
    region_id INT,
    FOREIGN KEY (region_id) REFERENCES region(id)
);
CREATE TABLE adresse (
    id INT PRIMARY KEY AUTO_INCREMENT,
    wilaya_id INT,
    info_supp TEXT,
    FOREIGN KEY (wilaya_id) REFERENCES wilaya (code)
);

CREATE TABLE theme (
    id INT PRIMARY KEY AUTO_INCREMENT,
    designation TEXT
);

CREATE TABLE categorie (
    id INT PRIMARY KEY AUTO_INCREMENT,
    designation TEXT
);

CREATE TABLE jour_choices (
    id INT PRIMARY KEY,
    label VARCHAR(50)
);

-- USERS
CREATE TABLE tourist_user (
    id INT PRIMARY KEY AUTO_INCREMENT,
    ntelephone BIGINT,
    adresse_id INT,
    username VARCHAR(150),
    password VARCHAR(128),
    nom VARCHAR(30),
    prenom VARCHAR(150),
    email VARCHAR(254),
    FOREIGN KEY (adresse_id) REFERENCES adresse (id)
);

CREATE TABLE regional_user (
    id INT PRIMARY KEY AUTO_INCREMENT,
    ntelephone BIGINT,
    region_id INT,
    username VARCHAR(150),
    password VARCHAR(128),
    nom VARCHAR(30),
    prenom VARCHAR(150),
    email VARCHAR(254),
    FOREIGN KEY (region_id) REFERENCES region (id)
);


--


CREATE TABLE type_event (
    id INT PRIMARY KEY AUTO_INCREMENT,
    designation TEXT
);


CREATE TABLE evenement (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nom TEXT,
    type_event_id INT,
    id_wilaya INT,
    FOREIGN KEY (id_wilaya) REFERENCES wilaya (code)
);

 
 
CREATE TABLE point_d_interet (
    id INT PRIMARY KEY AUTO_INCREMENT,
    description TEXT,
    nom TEXT,
    Dimanche TEXT,
	Lundi TEXT,
    Mardi TEXT,
    Mercredi TEXT,
    Jeudi TEXT,
    Vendredi TEXT,
    Samedi TEXT, 
    nbr_visites INT DEFAULT 0,
    moyenne_etoiles FLOAT DEFAULT 0,
    adresse_id INT,
    theme_id INT,
    categorie_id INT,
    lng FLOAT,
    lat FLOAT,
    FOREIGN KEY (adresse_id) REFERENCES adresse (id),
    FOREIGN KEY (theme_id) REFERENCES theme (id),
    FOREIGN KEY (categorie_id) REFERENCES categorie (id)
);



CREATE TABLE commentaire (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_touriste INT,
    contenu TEXT,
    nb_etoile INT,
    id_point_in INT,
    FOREIGN KEY (id_touriste) REFERENCES tourist_user (id),
    FOREIGN KEY (id_point_in) REFERENCES point_d_interet (id)
);


CREATE TABLE moyen_transport (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_point_d_interet INT,
    designation TEXT,
    FOREIGN KEY (id_point_d_interet) REFERENCES point_d_interet (id)
);

CREATE TABLE image (
  id INT AUTO_INCREMENT PRIMARY KEY,
  path VARCHAR(255),
  point_id INT,
  FOREIGN KEY (point_id) REFERENCES point_d_interet (id)
);

CREATE TABLE image_event (
  id INT AUTO_INCREMENT PRIMARY KEY,
  path VARCHAR(255),
  event_id INT,
  FOREIGN KEY (event_id) REFERENCES evenement (id)
);

INSERT INTO adresse (wilaya_id, info_supp)
SELECT w.code, CONCAT('Commune ', n.num, ' ',
    CASE FLOOR(RAND() * 5)
        WHEN 0 THEN 'Villa'
        WHEN 1 THEN 'Maison'
        WHEN 2 THEN 'Duplex'
        WHEN 3 THEN 'Chalet'
        WHEN 4 THEN 'Ferme'
    END
) AS info_supp
FROM wilaya w
JOIN (
    SELECT 1 AS num UNION ALL
    SELECT 2 AS num
) AS n
CROSS JOIN (
    SELECT 1 AS dummy UNION ALL
    SELECT 2 AS dummy
) AS d
ORDER BY w.code, RAND();


SELECT * FROM adresse
order by wilaya_id;

INSERT INTO tourist_user (ntelephone, adresse_id, username, password, nom, prenom, email)
SELECT
    FLOOR(RAND() * 900000000) + 1000000000 AS ntelephone,
    FLOOR(1 + FLOOR(RAND()) * (SELECT MAX(id) FROM adresse)) AS adresse_id,
    CONCAT(temp_nom.nom, CAST(FLOOR(RAND() * 100) AS UNSIGNED)) AS username,
    MD5(RAND()) AS password,
    temp_nom.nom AS nom,
    temp_prenom.prenom AS prenom,
    CONCAT(temp_nom.nom, temp_prenom.prenom, '@gmail.com') AS email
FROM
    (SELECT nom, RAND() AS rand_num FROM temp_nom ORDER BY rand_num LIMIT 50) AS temp_nom
JOIN
    (SELECT prenom, RAND() AS rand_num FROM temp_prenom ORDER BY rand_num LIMIT 50) AS temp_prenom
ON
    1 = 1;
